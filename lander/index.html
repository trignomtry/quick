QuickScript

Small, strongly typed, async-first, and LLVM-backed. Build servers, scripts, and services with predictable speed.

Read the Docs Quickstart 
let web = io.web()

io.listen(8087, fun(req: Request) {
    match req.path {
        "/hi"       => return web.text(200, "Hello!"),
        "/api/data" => return web.json("{\"message\":\"Hello!\"}"),
        _           => return web.text(404, "Not found"),
    }
})
